# right-censored model

with pm.Model() as m:
    beta0 = pm.Normal("beta0", 0, sigma=10)
    beta1 = pm.Normal("beta1", 0, sigma=10)
    
    位_uncensored = exp(beta0 + beta1 * x_uncensored)
    位_censored = exp(beta0 + beta1 * x_censored)
    
    likelihood_uncensored = pm.Exponential("likelihood_uncensored", 位_uncensored, observed=y_uncensored, shape=y_uncensored.shape[0])
    likelihood_censored = pm.Bound("censored", pm.Exponential.dist(位_censored), lower=censored, shape=censored.shape[0])

    mu_placebo = pm.Deterministic("mu_placebo", exp(-beta0))
    mu_chemo = pm.Deterministic("mu_chemo", exp(-beta0 - beta1))

    mu_diff = pm.Deterministic("mu_diff", mu_chemo - mu_placebo)

    H_prob = pm.Deterministic("H_prob", switch(ge(mu_diff, 0), 1, 0))

    trace = pm.sample(
        10000, tune=2000, cores=4, init="auto"
    )
